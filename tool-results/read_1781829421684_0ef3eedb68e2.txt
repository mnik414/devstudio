    36→function getInitials(name: string): string {
    37→  return name
    38→    .split(' ')
    39→    .filter(Boolean)
    40→    .slice(0, 2)
    41→    .map((n) => n[0]?.toUpperCase() ?? '')
    42→    .join('')

    56→function colorFromString(str: string): string {
    57→  let hash = 0
    58→  for (let i = 0; i < str.length; i++) hash = str.charCodeAt(i) + ((hash << 5) - hash)
    59→  return AVATAR_PALETTE[Math.abs(hash) % AVATAR_PALETTE.length]
    60→}

    70→function slugify(text: string): string {
    71→  return text
    72→    .toLowerCase()
    73→    .replace(/[^\w\s-]/g, '')
    74→    .trim()
    75→    .replace(/[\s_-]+/g, '-')
    76→    .replace(/^-+|-+$/g, '')
    77→}
    78→
    79→function extractText(node: ReactNode): string {
    80→  if (typeof node === 'string') return node
    81→  if (typeof node === 'number') return String(node)
    82→  if (Array.isArray(node)) return node.map(extractText).join('')
    83→  if (node && typeof node === 'object' && 'props' in node) {
    84→    // @ts-expect-error - react element shape
    85→    return extractText(node.props?.children)
    86→  }
    87→  return ''
    88→}

    90→function extractTOC(markdown: string): { id: string; text: string }[] {
    91→  const lines = markdown.split('\n')
    92→  const toc: { id: string; text: string }[] = []
    93→  let inCodeBlock = false
    94→  for (const line of lines) {
    95→    if (line.trim().startsWith('```')) {
    96→      inCodeBlock = !inCodeBlock
    97→      continue
    98→    }
    99→    if (inCodeBlock) continue
   100→    const match = line.match(/^##\s+(.+?)\s*$/)
   101→    if (match) {
   102→      const text = match[1].replace(/[*_`~]/g, '').trim()
   103→      toc.push({ id: slugify(text), text })
   104→    }
   105→  }
   106→  return toc
   107→}

   304→function TableOfContents({
   305→  items,
   306→  activeId,
   307→}: {
   308→  items: { id: string; text: string }[]
   309→  activeId: string
   310→}) {
   311→  const t = useT()
   312→  const handleClick = (e: React.MouseEvent<HTMLAnchorElement>, id: string) => {
   313→    e.preventDefault()
   314→    const el = document.getElementById(id)
   315→    if (el) {
   316→      el.scrollIntoView({ behavior: 'smooth', block: 'start' })
   317→      // update hash without jumping
   318→      if (typeof history !== 'undefined') history.replaceState(null, '', `#${id}`)
   319→    }
   320→  }
   321→
   322→  if (items.length === 0) return null
   323→
   324→  return (
   325→    <nav aria-label={t('blogDetail.contents')} className="text-sm">
   326→      <p className="mb-3 flex items-center gap-2 text-xs font-semibold uppercase tracking-wider text-muted-foreground">
   327→        <List className="h-3.5 w-3.5" />
   328→        {t('blogDetail.contents')}
   329→      </p>
   330→      <ul className="space-y-1 border-l border-border/60">
   331→        {items.map((item) => {
   332→          const isActive = activeId === item.id
   333→          return (
   334→            <li key={item.id} className="relative">
   335→              {isActive && (
   336→                <span
   337→                  className="absolute -left-px top-1/2 h-5 w-0.5 -translate-y-1/2 rounded-full bg-gradient-to-b from-primary to-accent"
   338→                  aria-hidden
   339→                />
   340→              )}
   341→              <a
   342→                href={`#${item.id}`}
   343→                onClick={(e) => handleClick(e, item.id)}
   344→                className={cn(
   345→                  '-ml-px block border-l-2 pl-4 py-1 transition-all duration-200',
   346→                  isActive
   347→                    ? 'border-transparent font-semibold text-foreground'
   348→                    : 'border-transparent text-muted-foreground hover:translate-x-0.5 hover:border-primary hover:text-primary',
   349→                )}
   350→              >
   351→                {item.text}
   352→              </a>
   353→            </li>
   354→          )
   355→        })}
   356→      </ul>
   357→    </nav>
   358→  )
   359→}

   464→/* ------------------------------ view ------------------------------ */
   465→
   466→export function BlogDetailView() {
   467→  const slug = useNav((s) => s.detailSlug)
   468→  const closeDetail = useNav((s) => s.closeDetail)
   469→  const setView = useNav((s) => s.setView)
   470→  const { data, isLoading, isError } = useBlogPost(slug)
   471→  const t = useT()
   472→  const lang = useLang((s) => s.lang)
   473→
   474→  const articleRef = useRef<HTMLElement>(null)
   475→  const item = data?.item
   476→  // Only track scroll when the article is actually rendered (avoids
   477→  // framer-motion "Target ref is defined but not hydrated" crash during loading)
   478→  const { scrollYProgress } = useScroll({
   479→    target: item ? articleRef : undefined,
   480→    offset: ['start start', 'end end'],
   481→  })
   482→  const scaleX = useSpring(scrollYProgress, {
   483→    stiffness: 100,
   484→    damping: 30,
   485→    restDelta: 0.001,
   486→  })
   487→
   488→  const related = data?.related ?? []
   489→  const content = item?.content ?? ''
   490→  const toc = useMemo(() => extractTOC(content), [content])
   491→  const [activeTocId, setActiveTocId] = useState<string>('')
   492→
   493→  // Track active section for TOC highlighting (only when there is content to track)
   494→  useEffect(() => {
   495→    if (toc.length === 0) return
   496→    const observer = new IntersectionObserver(
   497→      (entries) => {
   498→        const visible = entries
   499→          .filter((e) => e.isIntersecting)
   500→          .sort((a, b) => b.intersectionRatio - a.intersectionRatio)
   501→        if (visible[0]?.target.id) {
   502→          setActiveTocId(visible[0].target.id)
   503→        }
   504→      },
   505→      { rootMargin: '-20% 0px -65% 0px', threshold: [0, 0.25, 0.5, 1] },
   506→    )
   507→    toc.forEach((entry) => {
   508→      const el = document.getElementById(entry.id)
   509→      if (el) observer.observe(el)
   510→    })
   511→    return () => observer.disconnect()
   512→  }, [toc])
   513→
   514→  const handleBack = () => {
   515→    closeDetail()
   516→    setView('blog')
   517→  }
   518→
   519→  if (isLoading) {
   520→    return (
   521→      <div className="pt-8">
   522→        {/* progress placeholder */}
   523→        <div className="fixed left-0 right-0 top-16 z-40 h-1 bg-border/60" />
   524→        <ArticleSkeleton />
   525→      </div>
   526→    )
   527→  }
   528→
   529→  if (!item || isError) {
   530→    return (
   531→      <div className="pt-8">
   532→        <div className="fixed left-0 right-0 top-16 z-40 h-1 bg-border/60" />
   533→        <NotFound onBack={handleBack} />
   534→      </div>
   535→    )
   536→  }
   537→
   538→  return (
   539→    <>
   540→      {/* reading progress */}
   541→      <motion.div
   542→        style={{ scaleX }}
   543→        className="fixed left-0 right-0 top-16 z-50 h-1 origin-left bg-gradient-to-r from-primary to-accent"
   544→        aria-hidden
   545→      />
   546→
   547→      <article ref={articleRef} className="relative mx-auto max-w-6xl px-4 pb-24 pt-8 sm:px-6 lg:px-8">
   548→        {/* back button */}
   549→        <Reveal>
   550→          <button
   551→            type="button"
   552→            onClick={handleBack}
   553→            className="group mb-8 inline-flex items-center gap-1.5 text-sm font-medium text-muted-foreground transition-colors hover:text-foreground"
   554→          >
   555→            <ArrowLeft className={cn('h-4 w-4 transition-transform group-hover:-translate-x-0.5', lang === 'fa' && 'rtl-flip group-hover:translate-x-0.5')} />
   556→            {t('blogDetail.back')}
   557→          </button>
   558→        </Reveal>
   559→
